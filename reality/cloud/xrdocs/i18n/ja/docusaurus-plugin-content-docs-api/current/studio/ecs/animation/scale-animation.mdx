## 説明

このコンポーネントは、エンティティにアニメーションを作成します。

<ScaleAnimationTable />

## 機能

<ComponentFunctions
    name="ScaleAnimation"
    properties={{
        fromX: 0,
        fromY: 0,
        fromZ: 0,
        toX: 0,
        toY: 0,
        toZ: 0,
        autoFrom: false,
        duration: 1000,
        loop: true,
        reverse: false,
        easeIn: false,
        easeOut: false,
        easingFunction: "Quadratic"
    }}
    mutate={`cursor.toX = 2;

    cursor.toZ = 2;

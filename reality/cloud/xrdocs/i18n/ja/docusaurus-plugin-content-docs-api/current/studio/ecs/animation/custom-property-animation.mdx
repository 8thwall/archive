# カスタムプロパティアニメーション

## 説明

このコンポーネントは、エンティティにアニメーションを作成します。

## プロパティ

<CustomPropertyAnimationTable />

## 機能

<ComponentFunctions
    name="CustomPropertyAnimation"
    properties={{
        attribute: "attribute",
        property: "property",
        from: 0,
        to: 0,
        autoFrom: false,
        duration: 1000,
        loop: false,
        reverse: false,
        easeIn: false,
        easeOut: false,
        easingFunction: "Cubic"
    }}
